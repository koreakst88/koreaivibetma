**Обновить функцию isDayLocked:**
```javascript
// Проверка: заблокирован ли день
function isDayLocked(dayId) {
    // Публичные дни всегда открыты
    if (PUBLIC_DAYS.includes(dayId)) {
        return false;
    }
    
    // Остальные дни проверяем по кодам разблокировки
    const unlocked = getUnlockedDays();
    return !unlocked.includes(dayId);
}
```

**Добавить новую функцию:**
```javascript
// Проверка: публичный ли раздел
function isPublicSection(sectionId) {
    return PUBLIC_SECTIONS.includes(sectionId);
}

// Получить статус доступа к дню
function getDayAccessStatus(dayId) {
    if (PUBLIC_DAYS.includes(dayId)) {
        return {
            locked: false,
            reason: 'public',
            badge: 'Бесплатно'
        };
    }
    
    const unlocked = getUnlockedDays();
    if (unlocked.includes(dayId)) {
        return {
            locked: false,
            reason: 'unlocked',
            badge: null
        };
    }
    
    return {
        locked: true,
        reason: 'locked',
        badge: 'Для учеников курса'
    };
}
```

### 2. Обновление app.js (рендеринг главной)

**Обновить функцию renderDays:**
```javascript
function renderDays() {
    const container = document.getElementById('days-container');
    const progress = getProgress();
    
    Object.entries(DAYS_CONFIG).forEach(([dayId, dayInfo]) => {
        const accessStatus = getDayAccessStatus(dayId);
        const isCompleted = progress[dayId]?.completed || false;
        
        const card = document.createElement('div');
        card.className = `day-card ${accessStatus.locked ? 'locked' : 'unlocked'} ${isCompleted ? 'completed' : ''}`;
        
        if (accessStatus.locked) {
            // Заблокированный день
            card.innerHTML = `
                <div class="day-lock">🔒</div>
                <div class="day-info">
                    <h3>День ${dayInfo.order}: ${dayInfo.title}</h3>
                    <p class="day-duration">${dayInfo.duration}</p>
                    <span class="badge badge-locked">${accessStatus.badge}</span>
                </div>
                <button onclick="showEnrollmentInfo()" class="btn-enroll">
                    Записаться на курс
                </button>
            `;
        } else if (accessStatus.reason === 'public') {
            // Бесплатный день
            card.innerHTML = `
                <div class="day-status">🎁</div>
                <div class="day-info">
                    <h3>День ${dayInfo.order}: ${dayInfo.title}</h3>
                    <p class="day-duration">${dayInfo.duration}</p>
                    <span class="badge badge-free">${accessStatus.badge}</span>
                </div>
                <a href="day.html?id=${dayId}" class="btn-open">Открыть бесплатно</a>
            `;
        } else {
            // Разблокированный день (для учеников)
            card.innerHTML = `
                <div class="day-status">${isCompleted ? '✅' : '📖'}</div>
                <div class="day-info">
                    <h3>День ${dayInfo.order}: ${dayInfo.title}</h3>
                    <p class="day-duration">${dayInfo.duration}</p>
                </div>
                <a href="day.html?id=${dayId}" class="btn-open">Открыть</a>
            `;
        }
        
        container.appendChild(card);
    });
    
    updateProgressBar();
}
```

**Добавить функцию для показа информации о записи:**

```javascript
// Показать информацию о записи на курс
function showEnrollmentInfo() {
    const modal = document.createElement('div');
    modal.className = 'modal';
    modal.innerHTML = `
        <div class="modal-content">
            <h2>📚 Полный курс для учеников</h2>
            <p>Дни 1-7 доступны после записи на курс.</p>
            
            <h3>Что входит в полный курс:</h3>
            <ul>
                <li>7 индивидуальных встреч (60-90 минут)</li>
                <li>Доступ ко всем материалам навсегда</li>
                <li>Создание полноценного Telegram Mini App</li>
                <li>Поддержка преподавателя</li>
            </ul>
            
            <div class="modal-actions">
                <button onclick="contactTeacher()" class="btn-primary">
                    Записаться на курс
                </button>
                <button onclick="closeModal()" class="btn-secondary">
                    Закрыть
                </button>
            </div>
        </div>
    `;
    
    document.body.appendChild(modal);
    setTimeout(() => modal.classList.add('show'), 10);
}

function closeModal() {
    const modal = document.querySelector('.modal');
    modal.classList.remove('show');
    setTimeout(() => modal.remove(), 300);
}
```

```javascript
function updateDay0Status() {

    const statusElement = document.getElementById('day0-status');

        statusElement.textContent = '✅ Пройден';
```
